Shared Preferences

Shared Preferences란?​

Shared Preferences란 Swift에서 UserDefaults나 Android SharedPreference를 위한 플랫폼별 영구 저장소를 래핑한 라이브러리이며 데이터 베이스를 사용하지 않고 간단한 데이터를 저장할 때 사용합니다.
기본적으로 Key-Value의 Map 형식을 가지고 있습니다.

Shared Preferences 설치​

다른 권한 설정 없이 최신버전을 dependencies에 등록하기만 하면 사용할 수 있습니다.

dependencies:
  shared_preferences: ^2.0.15

Shared Preferences 사용​

인스턴스 선언​

final prefs = await SharedPreferences.getInstance();

데이터 저장​

prefs.setString('key', 'value');

데이터 읽기​

prefs.getString('key');

데이터 삭제하기​

prefs.remove('key');

예시​

import 'dart:async';

import 'package:flutter/material.dart';
import 'package:shared_preferences/shared_preferences.dart';

void main() {
  runApp(const MyApp());
}

class MyApp extends StatelessWidget {
  const MyApp({Key? key}) : super(key: key);

  @override
  Widget build(BuildContext context) {
    return const MaterialApp(
      title: 'SharedPreferences Demo',
      home: SharedPreferencesDemo(),
    );
  }
}

class SharedPreferencesDemo extends StatefulWidget {
  const SharedPreferencesDemo({Key? key}) : super(key: key);

  @override
  SharedPreferencesDemoState createState() => SharedPreferencesDemoState();
}

class SharedPreferencesDemoState extends State<SharedPreferencesDemo> {
  // SharedPreferences 선언
  final Future<SharedPreferences> _prefs = SharedPreferences.getInstance();
  late Future<int> _counter;

  // 현재 SharedPreferences 저장된 counter 값이 없으면 0으로 초기화 해주고 있다면 해당 값을 가져와서 1을 더해줌
  // 방금 더 해준 값을 SharedPreferences 인스턴스 'counter'에 저장함
  Future<void> _incrementCounter() async {
    final SharedPreferences prefs = await _prefs;
    final int counter = (prefs.getInt('counter') ?? 0) + 1;

    setState(() {
      _counter = prefs.setInt('counter', counter).then((bool success) {
        return counter;
      });
    });
  }

  @override
  void initState() {
    super.initState();
    _counter = _prefs.then((SharedPreferences prefs) {
      return prefs.getInt('counter') ?? 0;
    });
  }

  @override
  Widget build(BuildContext context) {
    return Scaffold(
      appBar: AppBar(
        title: const Text('SharedPreferences Demo'),
      ),
      body: Center(
          child: FutureBuilder<int>(
              future: _counter,
              builder: (BuildContext context, AsyncSnapshot<int> snapshot) {
                switch (snapshot.connectionState) {
                  case ConnectionState.waiting:
                    return const CircularProgressIndicator();
                  default:
                    if (snapshot.hasError) {
                      return Text('Error: ${snapshot.error}');
                    } else {
                      return Text(
                        'Button tapped ${snapshot.data} time${snapshot.data == 1 ? '' : 's'}.\n\n'
                            'This should persist across restarts.',
                      );
                    }
                }
              })),
      floatingActionButton: FloatingActionButton(
        onPressed: _incrementCounter,
        tooltip: 'Increment',
        child: const Icon(Icons.add),
      ),
    );
  }
}
